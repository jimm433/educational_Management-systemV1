Q15：(10分) 撰寫程式
try
{
    StreamWriter writer = File.CreateText("multiples_of_three.txt");
    
    for (int i = 1; i <= 999; i++)
    {
        if (i % 3 == 0)
        {
            writer.WriteLine(i);
        }
    }
    
    writer.Close();
    MessageBox.Show("檔案寫入完成");
}
catch (Exception ex)
{
    MessageBox.Show("檔案寫入錯誤：" + ex.Message);
}

private void SetToZero(out int number)
{
    number = 0;
}

{

    outputListBox.Items.Add("y is set to " + y);
